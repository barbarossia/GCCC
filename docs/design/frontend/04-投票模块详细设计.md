# 投票模块详细设计文档

## 1. 模块概述

投票模块是社区治理的核心执行组件，负责处理用户对提案的投票操作。用户需要质押 GCCC Token 来获得投票权，投票权重根据质押数量和时间计算。该模块确保投票过程的透明性和公平性。

## 2. 功能需求

### 2.1 核心功能
- **提案展示**: 显示处于投票阶段的提案详情
- **质押管理**: 质押/取消质押 GCCC Token
- **投票操作**: 对提案进行投票
- **投票权计算**: 基于质押数量和时间的投票权重
- **实时统计**: 显示投票进度和统计数据
- **投票历史**: 查看用户的投票记录

### 2.2 投票规则
- **质押要求**: 必须质押 GCCC Token 才能获得投票权
- **投票权重**: 质押数量 × 质押时间权重
- **单次投票**: 每个提案只能投票一次
- **投票期限**: 仅在投票阶段可以投票 (每月11-19日)
- **防闪电贷**: 质押时间最短要求防止攻击

### 2.3 投票权重计算
```
投票权重 = 质押数量 × 时间权重

时间权重规则:
- 质押 < 1天: 0.1x
- 质押 1-7天: 0.5x  
- 质押 7-30天: 0.8x
- 质押 > 30天: 1.0x
```

```typescript
            <div className="progress-bar">

        <VoteConfirmDialog
          proposal={proposal}
          votingPower={userVotingPower}
          onConfirm={onVoteSuccess}
          onCancel={() => setShowVoteDialog(false)}
        />
      )}
    </>

// 投票权重计算函数
const calculateVotingPower = (stakedAmount: number, stakingDuration: number): number => {
  let timeWeight = 0.1; // 默认权重
  
  if (stakingDuration >= 30) {
    timeWeight = 1.0;
  } else if (stakingDuration >= 7) {
    timeWeight = 0.8;
  } else if (stakingDuration >= 1) {
    timeWeight = 0.5;
  }
  
  return Math.floor(stakedAmount * timeWeight);
};
```

```typescript
// components/voting/VoteConfirmDialog.tsx
import React, { useState } from 'react';
import { useWallet } from '@solana/wallet-adapter-react';
import { voteForProposal } from '../../services/votingService';
import { Proposal } from '../../types/voting';
import { formatNumber } from '../../utils/formatters';
import { toast } from 'react-toastify';

interface VoteConfirmDialogProps {
  proposal: Proposal;
  votingPower: number;
  onConfirm: () => void;
  onCancel: () => void;
}

export const VoteConfirmDialog: React.FC<VoteConfirmDialogProps> = ({
  proposal,
  votingPower,
  onConfirm,
  onCancel
}) => {
  const { publicKey, signTransaction } = useWallet();
  const [voting, setVoting] = useState(false);

  const handleVote = async () => {
    if (!publicKey || !signTransaction) {
      toast.error('钱包未连接');
      return;
    }

    setVoting(true);
    try {
      await voteForProposal({
        proposalId: proposal.id,
        voter: publicKey.toString(),
        votingPower,
        signTransaction
      });

      toast.success('投票成功!');
      onConfirm();
    } catch (error) {
      console.error('投票失败:', error);
      toast.error('投票失败，请重试');
    } finally {
      setVoting(false);
    }
  };

  return (
    <div className="vote-confirm-dialog-overlay">
      <div className="vote-confirm-dialog">
        <div className="dialog-header">
          <h3>确认投票</h3>
          <button 
            className="close-btn"
            onClick={onCancel}
          >
            ×
          </button>
        </div>

        <div className="dialog-content">
          {/* 提案信息 */}
          <div className="proposal-summary">
            <div className="proposal-image">
              <img 
                src={proposal.imageUrl || '/placeholder-coin.svg'} 
                alt={proposal.title}
              />
            </div>
            <div className="proposal-info">
              <h4>{proposal.title}</h4>
              <p className="event-name">{proposal.eventName}</p>
            </div>
          </div>

          {/* 投票详情 */}
          <div className="vote-details">
            <div className="detail-row">
              <span className="label">您的投票权:</span>
              <span className="value">{formatNumber(votingPower, 0)} GCCC</span>
            </div>
            
            <div className="detail-row">
              <span className="label">当前支持票数:</span>
              <span className="value">{formatNumber(proposal.voteCount, 0)} GCCC</span>
            </div>
            
            <div className="detail-row highlight">
              <span className="label">投票后支持票数:</span>
              <span className="value">
                {formatNumber(proposal.voteCount + votingPower, 0)} GCCC
              </span>
            </div>
          </div>

          {/* 投票说明 */}
          <div className="vote-warning">
            <div className="warning-icon">⚠️</div>
            <div className="warning-text">
              <p>请注意:</p>
              <ul>
                <li>每个提案只能投票一次</li>
                <li>投票后无法撤销</li>
                <li>投票期结束前可以取消质押</li>
              </ul>
            </div>
          </div>
        </div>

        <div className="dialog-actions">
          <button 
            className="cancel-btn"
            onClick={onCancel}
            disabled={voting}
          >
            取消
          </button>
          <button 
            className="confirm-btn primary"
            onClick={handleVote}
            disabled={voting}
          >
            {voting ? '投票中...' : '确认投票'}
          </button>
        </div>
      </div>
    </div>
  );
};
```

```typescript
// components/voting/VotingProgress.tsx
import React from 'react';
import { motion } from 'framer-motion';

            <span className="value">{participationRate.toFixed(1)}%</span>

// 投票倒计时组件
const VotingCountdown: React.FC<{ endTime: Date }> = ({ endTime }) => {
  const [timeLeft, setTimeLeft] = useState(calculateTimeLeft());

  function calculateTimeLeft() {
    const difference = +endTime - +new Date();
    let timeLeft = {};

    if (difference > 0) {
      timeLeft = {
        days: Math.floor(difference / (1000 * 60 * 60 * 24)),
        hours: Math.floor((difference / (1000 * 60 * 60)) % 24),
        minutes: Math.floor((difference / 1000 / 60) % 60),
        seconds: Math.floor((difference / 1000) % 60)
      };
    }

    return timeLeft;
  }

  useEffect(() => {
    const timer = setTimeout(() => {
      setTimeLeft(calculateTimeLeft());
    }, 1000);

    return () => clearTimeout(timer);
  });

  return (
    <div className="countdown-display">
      {Object.keys(timeLeft).length > 0 ? (
        <div className="countdown-units">
          <div className="unit">
            <span className="number">{timeLeft.days || 0}</span>
            <span className="label">天</span>
          </div>
          <div className="unit">
            <span className="number">{timeLeft.hours || 0}</span>
            <span className="label">时</span>
          </div>
          <div className="unit">
            <span className="number">{timeLeft.minutes || 0}</span>
            <span className="label">分</span>
          </div>
          <div className="unit">
            <span className="number">{timeLeft.seconds || 0}</span>
            <span className="label">秒</span>
          </div>
        </div>
      ) : (
        <div className="countdown-ended">投票已结束</div>
      )}
    </div>
  );
};
```

```typescript
// services/votingService.ts
import { Connection, PublicKey, Transaction } from '@solana/web3.js';
import { Program, BN } from '@project-serum/anchor';
import { getGovernanceProgram } from '../programs/governance';

export interface VoteRequest {
  proposalId: string;
  voter: string;
  votingPower: number;
  signTransaction: (transaction: Transaction) => Promise<Transaction>;
}

export const voteForProposal = async (request: VoteRequest): Promise<string> => {
  const connection = new Connection(process.env.NEXT_PUBLIC_SOLANA_RPC_ENDPOINT);
  const program = getGovernanceProgram(connection);
  
  const voterPublicKey = new PublicKey(request.voter);
  const proposalPublicKey = new PublicKey(request.proposalId);

  // 构建投票交易
  const transaction = await program.methods
    .vote(new BN(request.votingPower))
    .accounts({
      proposal: proposalPublicKey,
      voter: voterPublicKey,
      voterTokenAccount: await getAssociatedTokenAddress(
        GCCC_TOKEN_MINT,
        voterPublicKey
      ),
      governanceProgram: program.programId,
    })
    .transaction();

  // 用户签名
  const signedTransaction = await request.signTransaction(transaction);
  
  // 发送交易
  const signature = await connection.sendRawTransaction(
    signedTransaction.serialize()
  );
  
  // 确认交易
  await connection.confirmTransaction(signature, 'confirmed');
  
  return signature;
};

export const stakeTokens = async (
  amount: number,
  publicKey: PublicKey,
  signTransaction: (transaction: Transaction) => Promise<Transaction>
): Promise<string> => {
  const connection = new Connection(process.env.NEXT_PUBLIC_SOLANA_RPC_ENDPOINT);
  const program = getGovernanceProgram(connection);

  const transaction = await program.methods
    .stake(new BN(amount * 1e9)) // 转换为最小单位
    .accounts({
      user: publicKey,
      userTokenAccount: await getAssociatedTokenAddress(
        GCCC_TOKEN_MINT,
        publicKey
      ),
      stakingPool: STAKING_POOL_ADDRESS,
      tokenProgram: TOKEN_PROGRAM_ID,
    })
    .transaction();

  const signedTransaction = await signTransaction(transaction);
  const signature = await connection.sendRawTransaction(
    signedTransaction.serialize()
  );
  
  await connection.confirmTransaction(signature, 'confirmed');
  return signature;
};
```

```typescript
const validateVote = async (
  proposalId: string,
  voter: PublicKey
): Promise<boolean> => {
  // 检查投票期
  const isVotingPeriod = await checkVotingPeriod();
  if (!isVotingPeriod) return false;

  // 检查是否已投票
  const hasVoted = await checkUserVoted(proposalId, voter);
  if (hasVoted) return false;

  // 检查质押状态
  const stakingInfo = await getStakingInfo(voter);

  return true;
};
```
